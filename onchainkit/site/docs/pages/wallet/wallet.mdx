# `<Wallet />`

:::warning
Component is actively in development. Stay tuned for upcoming releases.
:::

## Usage

```tsx
import { Avatar, Name, Identity } from '@coinbase/onchainkit/identity';
import {
  ConnectAccount,
  EthBalance, 
  Wallet, 
  WalletDropdown, 
  WalletDropdownItem, 
  WalletDropdownDisconnect
} from '@coinbase/onchainkit/wallet';

<Wallet>
  <ConnectAccount>
    <Identity>
      <Avatar />
      <Name />
    </Identity>
  </ConnectAccount>
  <WalletDropdown>
    <WalletDropdownItem>
      <Identity>
        <Avatar />
        <Name showAttestation />
        <Name showAddress />
        <EthBalance>
      </Identity>
    </WalletDropdownItem>
    <WalletDropdownItem>
      <a href="link">Go to Wallet Dashboard<a>
    </WalletDropdownItem>
    <WalletDropdownDisconnect />
  </WalletDropdown>
</Wallet>
```
